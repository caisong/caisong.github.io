    'DRIVER={SQL Server};SERVER=10.167.219.84;PORT=1433;\
    DATABASE=ITPDB_41185;UID=sa;PWD=Sa123456')

cursor = cnxn.cursor()

for i in range(1, 6001):
    try:
        cursor.execute('insert into tb_MaintenanceRecv  (SerialNumber, CarNumber, \
        ResponseStatus) values (?, ?, ?)',

gitbook 安装流程

mkdir -p ~/.nvm/versions/node
tar xvf node-v12.22.12-linux-x64.tar.xz
mv node-v12.22.12-linux-x64 ~/.nvm/versions/node/v12.22.12

npm config set registry https://registry.npmmirror.com

npm install -g gitbook-cli

gitbook init
gitbook build
gitbook install
